33 /**
34  * @brief Interpolate solutions fields from one finite element mesh to another.
35  *
36  * Unlike the dealii function VectorTools::interpolate_to_different_mesh, this function
37  * doesn't assume that the parallel partitioning of the two meshes are same.
38  * Further the two meshes can be arbitraririly refined. The only constraint is that they
39  * must discretize the same real-space domain.
40  *
41  * @author Sambit Das
42  */

53  /**
54  * @brief Projects a vector of parallel distributed vectors
55  * from previous to current mesh.
56  *
57  * triangulationSerPrev and triangulationParPrev are serial and parallel versions of
58  * the previous mesh triangulation. Currently we need triangulationSerPrev for the algorithm to work,
59  * but the algorithm could be potentially improved to not need triangulationSerPrev. When
60  * that happens, we can directly pass previous and current dofHandlers as input arguments.
61  *
62  * @param triangulationSerPrev serial triangulation of previous mesh
63  * @param triangulationParPrev parallel distributed triangulation of previous mesh
64  * @param triangulationParCurrent parallel distributed triangulation of current mesh
65  * @param FEPrev FiniteElement object of the previous mesh
66  * @param FECurrent FiniteElement object of the current mesh. FECurrent and FEPrev must have
67  * the same number of components.
68  * @param fieldsPreviousMesh parallel distributed fields on previous mesh to be interpolated from
69  * @param fieldsCurrentMesh parallel distributed fields on current mesh to be interpolated upon
70  * @param constraintsCurrentPtr pointer to ConstraintMatrix object of current mesh. Default value
71  * is NULL in which case the distribute operation doesn't happen inside interpolate. We have this
72  * function so that outside interpolate function we can use the inhouse distribute function.
73  */
74  void interpolate(const dealii::parallel::distributed::Triangulation<3> & triangulationSerPrev,

77  const dealii::FESystem<3> & FEPrev,
78  const dealii::FESystem<3> & FECurrent,
79  const std::vector<vectorType*> & fieldsPreviousMesh,
80  std::vector<vectorType*> & fieldsCurrentMesh,
81  const dealii::ConstraintMatrix * constraintsCurrentPtr=NULL);
